DROP TABLE IF EXISTS player_stats CASCADE;
DROP TABLE IF EXISTS players CASCADE;
DROP TABLE IF EXISTS matches CASCADE;
DROP TABLE IF EXISTS managers CASCADE;
DROP TABLE IF EXISTS teams CASCADE;

CREATE TABLE teams (
    team_id SERIAL PRIMARY KEY,
    team_name VARCHAR(100) NOT NULL,
    team_founded_year INT,
    team_home_stadium VARCHAR(100) NOT NULL,
    team_matches_played INT DEFAULT 0 CHECK (team_matches_played >= 0),
    team_matches_won INT DEFAULT 0 CHECK (team_matches_won >= 0),
    team_matches_drawn INT DEFAULT 0 CHECK (team_matches_drawn >= 0),
    team_matches_lost INT DEFAULT 0 CHECK (team_matches_lost >= 0),
    team_goals_scored INT DEFAULT 0 CHECK (team_goals_scored >= 0),
    team_goals_conceded INT DEFAULT 0 CHECK (team_goals_conceded >= 0),
    team_goal_difference INT DEFAULT 0,
    team_points INT DEFAULT 0 CHECK (team_points >= 0)
);

CREATE TABLE players (
    player_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    player_dob DATE,
    player_nationality VARCHAR(50),
    position VARCHAR(30),
    team_id INT,
    CONSTRAINT fk_team FOREIGN KEY (team_id) REFERENCES Teams(team_id) ON DELETE SET NULL
);

CREATE TABLE matches (
    match_id SERIAL PRIMARY KEY,
    home_team_id INT NOT NULL,
    away_team_id INT NOT NULL,
    match_date DATE NOT NULL,
    competition VARCHAR(50),
    home_team_score INT,
    away_team_score INT,
    CONSTRAINT fk_home_team FOREIGN KEY (home_team_id) REFERENCES Teams(team_id) ON DELETE RESTRICT,
    CONSTRAINT fk_away_team FOREIGN KEY (away_team_id) REFERENCES Teams(team_id) ON DELETE RESTRICT
);

CREATE TABLE player_stats (
    stat_id SERIAL PRIMARY KEY,
    player_id INT NOT NULL,
    match_id INT NOT NULL,
    player_goals INT DEFAULT 0,
    player_assists INT DEFAULT 0,
    player_minutes INT,
    player_yellow_cards INT DEFAULT 0,
    player_red_cards INT DEFAULT 0,
    CONSTRAINT fk_player FOREIGN KEY (player_id) REFERENCES Players(player_id) ON DELETE CASCADE,
    CONSTRAINT fk_match FOREIGN KEY (match_id) REFERENCES Matches(match_id) ON DELETE CASCADE
);

CREATE TABLE managers (
    manager_id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    manager_nationality VARCHAR(50),
    team_id INT,
    manager_hire_date DATE,
    CONSTRAINT fk_team_coach FOREIGN KEY (team_id) REFERENCES Teams(team_id) ON DELETE SET NULL
);
